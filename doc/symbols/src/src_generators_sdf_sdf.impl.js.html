  5 /**
  6  * Returns 1 if x is positive, 0 otherwise
  7  * @param {*} x 
  8  */
  9 function sign(x) { 
 10     if(x === 0)
 11         return 1;
 12     else {
 13         return x/Math.abs(x);
 14     }
 15 }

 20 /**
 21  *  Returns an array T2 where T2[i] = k*T[i]
 22  * @param {Array} T 
 23  * @param {Number} k 
 24  * @returns 
 25  */
 26 function tabMultiply(T,k) { 
 27     let T2 = new Array(T.length);
 28     for(let i = 0; i < T.length; i++) {
 29         T2[i] = T[i]*k;
 30     }
 31     return T2;
 32 }

 36 /**
 37  * Returns an image of a circle drawn with signed distance
 38  * @param {Object} args : width, height and r the radius of the circle are used.
 39  */
 40 function sdCircle(args) {
 41 
 42     if(typeof args.r === undefined) { args.r = 80;}
 43 
 44     
 45     let image = createImageData(args.width, args.height);
 46 
 47     let w = args.width;
 48     let h = args.height;
 49     let n = 0;
 50 
 51     for(let y = 0; y < args.height; y++){
 52 
 53         for(let x = 0; x < args.width; x++, n+=4){
 54 
 55             let k = Math.sqrt((w/2-x)**2 + (h/2-y)**2) - (w+h)/8; //The signed distance of the point to the center of the image
 56 
 57             // Computation of the colors to show
 58             let c = [1-sign(k)*0.2,1-sign(k)*0.5,1-sign(k)*0.6];
 59             c = tabMultiply(c,1-Math.exp(-0.05*Math.abs(k)));
 60             c = tabMultiply(c,0.8+0.2*Math.cos(k/2));
 61 
 62             image.data[n] = c[0]*255;
 63             image.data[n+1] = c[1]*255;
 64             image.data[n+2] = c[2]*255;
 65             image.data[n+3] = 255;
 66         }
 67     }
 68     return image;
 69 }

 74 /**
 75  * Returns an image of a rectangle drawn with signed distance
 76  * @param {Object} args : width, height and p are used.
 77  */
 78 function sdRectangle(args) {
 79 
 80     if(typeof args.p === undefined) { args.p = 5;}
 81     
 82     let image = createImageData(args.width, args.height);
 83     
 84     let w = args.width;
 85     let h = args.height;
 86     let n = 0;
 87 
 88     for(let y = 0; y < args.height; y++){
 89 
 90         for(let x = 0; x < args.width; x++, n+=4){
 91 
 92             let xc = x - w/2;  //The coordonates of the point in the system 
 93             let yc = y - h/2;  //centered at the center of the image
 94 
 95             let k = Math.max(w/h*Math.abs(yc),Math.abs(xc)) - (w+h)/8 ; // The signed distance of the point to the center of the image, corresponding to a rectangle
 96 
 97             // Computation of the colors to show
 98             let c = [1-sign(k)*0.2,1-sign(k)*0.5,1-sign(k)*0.6];
 99             c = tabMultiply(c,1-Math.exp(-0.05*Math.abs(k)));
100             c = tabMultiply(c,0.8+0.2*Math.cos(k/2));
101 
102             image.data[n] = c[0]*255;
103             image.data[n+1] = c[1]*255;
104             image.data[n+2] = c[2]*255;
105             image.data[n+3] = 255;
106         }
107     }
108     return image;

114 /**
115  * Returns an image of a star drawn with signed distance
116  * @param {Object} args : width, height, p and r are used.
117  * @returns 
118  */
119 function sdStar(args) {
120 
121     if(typeof args.p === undefined) { args.p = 5;}
122     if(typeof args.r === undefined) { args.r = 80;}
123 
124     
125     let image = createImageData(args.width, args.height);
126 
127     let w = args.width;
128     let h = args.height;
129     let an = Math.PI/args.p; // The angle of a section
130     let n = 0;
131 
132     for(let y = 0; y < args.height; y++){
133 
134         for(let x = 0; x < args.width; x++, n+=4){
135 
136             let xc = x - w/2; //The coordinates of the point in the system 
137             let yc = y - h/2; //centered at the center of the image
138 
139 
140             // Computation of the angle of the point in the system
141             let a = sign(yc)*Math.PI/2;
142 
143             if(xc > 0) {
144                 a = Math.atan(yc/xc);
145             }
146             else if (xc < 0) {
147                 a = Math.atan(yc/xc) + Math.PI;
148             }
149 
150             a += Math.PI/2;
151 
152             //Local angle in the corresponding section
153             let ac = a%(2*an);
154 
155             
156             let d = ((w+h)/8-args.r)*Math.abs(ac-an)/an+args.r;
157             
158             //Signed distance corresponding to a star
159             let k = Math.cos(a-(Math.trunc(a/(2*an))*2*an+an))*Math.sqrt((w/2-x)**2 + (h/2-y)**2) - d;
160 
161                         
162             // Computation of the colors to show
163             let c = [1-sign(k)*0.2,1-sign(k)*0.5,1-sign(k)*0.6];
164             c = tabMultiply(c,1-Math.exp(-0.05*Math.abs(k)));
165             c = tabMultiply(c,0.8+0.2*Math.cos(k/2));
166             
167             image.data[n] = c[0]*255;
168             image.data[n+1] = c[1]*255;
169             image.data[n+2] = c[2]*255;
170             image.data[n+3] = 255;
171         }
172     }
173     return image;
174 }
