  3 /*
  4  * Add two images (first image is overwritten)
  5  * args entries : width, height
  6  */
  7 function add(args, imgs) {
  8     let img1 = imgs[0];
  9     let img2 = imgs[1];
 10     let n = 0;
 11     for (let x = 0; x < args.height; x++) {
 12 	for (let y = 0; y < args.width; y++, n+= 4) {
 13             for (let i = 0; i < 4; i++) {
 14                 img1.data[n+i] = Math.min(255, img1.data[n+i]+img2.data[n+i]);
 15             }
 16         }
 17     }
 18     return img1;
 19 }
