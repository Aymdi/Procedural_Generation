1 const List = require('./list.impl.js');
  2 const Tree = require('./tree.impl.js');
  3 
  4 /*
  5  * Textures are trees that contain filter and and generator nodes.
  6  * The first child of a node (either a filter or a generator) is a dictionary which contains the arguments for that node.
  7  * Generator nodes only have one child - their args dictionary.
  8  * Filter nodes have two: their args dictionary, and an array of generator nodes that they take as arguments to produce a new image.
  9  */
 10 
 11 
 12 // a texture whose root is a generator function
 13 function genNode(gen, args={}) {
 14     return Tree.node(gen, List.cons(args, List.nil));
 15 }
 16 
 17 // a texture whose root is a filter function
 18 function filtNode(filter, genNodes, args={}) {
 19     return Tree.node(filter, List.cons(args, List.cons(genNodes, List.nil)));
 20 }
 21 
 22 // compute a texture, going bottom-up
 23 function makeTexture(texture, width=100, height=100) {
 24     let l = Tree.children(texture);
 25     if (List.isEmpty(List.tail(l))) { // case: texture is a generator
 26 	let gen = Tree.val(texture);
 27 	let args = List.head(l);
 28 	args.width = width;
 29 	args.height = height;
 30 	return gen(args);
 31     } else { // case: texture is a filter
 32 	let filter = Tree.val(texture);
 33 	let args = List.head(l);
 34 	args.width = width;
 35 	args.height = height;
 36 	let genNodes = List.head(List.tail(l));
 37 	let sourceImgs = [];
 38 	for (let i = 0; i < genNodes.length; i++) {
 39             sourceImgs[i] = makeTexture(genNodes[i], width, height);
 40 	}
 41 	return filter(args, sourceImgs);
 42     }
 43 }
 44 
 45 
 46 exports.makeGen = genNode;
 47 exports.makeFilter = filtNode;
 48 exports.makeTexture = makeTexture;
