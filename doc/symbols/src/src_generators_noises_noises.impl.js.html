 17 /**
 18  * gradient function
 19  * @param {*} h 
 20  * @param {*} x 
 21  * @param {*} y 
 22  */
 23 function gradient(h,x,y) {
 24     h = h%4;
 25     if(h === 0)
 26         return x+y;
 27     else if(h === 1)
 28         return -x+y;
 29     else if(h === 2)
 30         return x-y;
 31     return -x-y;
 32 }

 34 /**
 35  * interpolation function
 36  * @param {*} bl 
 37  * @param {*} br 
 38  * @param {*} tl 
 39  * @param {*} tr 
 40  * @param {*} x 
 41  * @param {*} y 
 42  * @param {*} f 
 43  */
 44 function interpolation(bl,br,tl,tr,x,y,f) {
 45     return f(1-x)*f(y)*bl+f(x)*f(y)*br+f(1-x)*f(1-y)*tl+f(x)*f(1-y)*tr;
 46 }

 50  * @param {*} x 
 51  */
 52 function smoothstep(x) {

 56 /**
 57  * perlinNoise function
 58  * @param {Object} args : p, color1 and color2 are used.
 59  */
 60 function perlinNoise(args) {
 61 
 62     if(typeof args.p === undefined) { args.p = 10;}
 63     if(typeof args.color1 === undefined) { args.color1 = [0,0,0,255];}
 64     if(typeof args.color2 === undefined) { args.color2 = [255,255,255,255];}
 65     
 66     let image = createImageData(args.width, args.height);
 67     let t = createPerlinTab();
 68     let n = 0;
 69     for(let y = 0; y < args.height; y++) {
 70         for(let x = 0; x < args.width; x++, n+=4) {
 71             let xi = x/(args.width/args.p);
 72             let yi = y/(args.height/args.p);
 73             let x0 = Math.floor(xi)%256;
 74             let y0 = Math.floor(yi)%256;
 75             let xc = xi-Math.floor(xi);
 76             let yc = yi-Math.floor(yi);
 77             let tl = t[t[x0]+y0];
 78             let tr = t[t[x0+1]+y0];
 79             let bl = t[t[x0]+y0+1];
 80             let br = t[t[x0+1]+y0+1];
 81             let k = interpolation(gradient(bl,xc,yc-1), gradient(br,xc-1,yc-1), gradient(tl,xc,yc), gradient(tr,xc-1,yc), xc, yc, smoothstep);
 82             k=(k+1)/2;
 83             image.data[n] = args.color1[0] + k*(args.color2[0]-args.color1[0]);
 84             image.data[n+1] = args.color1[1] + k*(args.color2[1]-args.color1[1]);
 85             image.data[n+2] = args.color1[2] + k*(args.color2[2]-args.color1[2]);
 86             image.data[n+3] = 255;
 87         }
 88     }
 89     return image;
 90 }

102 function fractalNoise(args) {
103 
104     if(typeof args.p === undefined) { args.p = 10;}
105     if(typeof args.color1 === undefined) { args.color1 = [0,0,0,255];}
106     if(typeof args.color2 === undefined) { args.color2 = [255,255,255,255];}
107     
108     let image = createImageData(args.width, args.height);
109     let t = createPerlinTab();
110     let n = 0;
111 
112     for(let y = 0; y < args.height; y++) {
113         for(let x = 0; x < args.width; x++, n+=4) {
114 
115             let freq = 2;
116             let ampl = 0.5;
117             let k = 0;
118 
119             for(let i = 0; i < args.p; i++) {
120                 let xi = x/(args.width/freq);
121                 let yi = y/(args.height/freq);
122                 let x0 = Math.floor(xi)%256;
123                 let y0 = Math.floor(yi)%256;
124                 let xc = xi-Math.floor(xi);
125                 let yc = yi-Math.floor(yi);
126                 let tl = t[t[x0]+y0];
127                 let tr = t[t[x0+1]+y0];
128                 let bl = t[t[x0]+y0+1];
129                 let br = t[t[x0+1]+y0+1];
130                 k += ampl*interpolation(gradient(bl,xc,yc-1), gradient(br,xc-1,yc-1), gradient(tl,xc,yc), gradient(tr,xc-1,yc), xc, yc, smoothstep);
131                 freq *= 2;
132                 ampl /= 2;
133             }
134 
135             k=(k+1)/2;
136 
137             image.data[n] = args.color1[0] + k*(args.color2[0]-args.color1[0]);
138             image.data[n+1] = args.color1[1] + k*(args.color2[1]-args.color1[1]);
139             image.data[n+2] = args.color1[2] + k*(args.color2[2]-args.color1[2]);
140             image.data[n+3] = 255;
141         }
142     }
