  3 /**
  4  * convolutionKer 
  5  */
  6 function convolutionKer(){
  7 	return [-1,-1,-1,-1,8,-1,-1,-1,-1];
  8 }
  9 
 10 ////////// Computing the new value of the current pixel //////////
 11 
 12 /**
 13  * newValue
 14  * @param {*} ker 
 15  * @param {*} pixels 
 16  * @param {*} n 
 17  * @returns {*} {R,G,B,A}
 18  */
 19 function newValue(ker,pixels,n){
 20 	const m = Math.trunc(n/2)+n;
 21 	let newVal = {R:0,G:0,B:0,A:pixels[m]};
 22 	for (let i=0 ; i < n*n ; ++i){
 23 		const R = pixels[i];
 24 		const G = pixels[i+1];
 25 		const B = pixels[i+2];
 26 		newVal['R'] += ker[i]*R;
 27 		newVal['G'] += ker[i]*G;
 28 		newVal['B'] += ker[i]*B;
 29 	}
 30 	return newVal;
 31 }
 32 
 33 ////////// Extracting the local pixels //////////
 34 
 35 /**
 36  * extractPixels
 37  * @param {*} t 
 38  * @param {*} i 
 39  * @param {*} j 
 40  * @param {*} n 
 41  * @param {*} height 
 42  * @param {*} width  
 43  */
 44 function extractPixels(t,i,j,n,height,width){
 45 	let pixels = new Array();
 46     //console.log(n);
 47 	for (let k=-n ; k < n ; ++k){
 48 		for (let l=-n ; l < n ; ++l){
 49 			const p = i+k;
 50 			const q = j+l;
 51 			if ( (p < 0) || (p >= width) || (q < 0) || (q >= height) ){
 52 				const m = 4*(i*width+j);
 53 				const newElement = t.slice(m,m+4);
 54 				pixels.push(newElement);
 55 			}
 56 
 57 			else {
 58 				const m = 4*(p*width + q);
 59 				const newElement = t.slice(m,m+4);
 60 				pixels.push(newElement);
 61 			}
 62 			//console.log(pixels.length);
 63 		}
 64 	}
 65 	return pixels;  
 66 }

 68 ////////// The main function //////////

 70 /**
 71  * convolutionFilter
 72  * @param {Object} args 
 73  * @param {Array} imgs 
 74  */
 75 function convolutionFilter(args,imgs){
 76 	const t = imgs[0].data;
 77 	//const width = args.width
 78 	//const height = args.height
 79 	let n = 0; // Index inside the image array
 80 	const newImage = t.slice();
 81 	const ker = convolutionKer();
 82 	const dimKer = ker.length;
 83 	for (let i = 0; i < args.height; i++) {
 84 		for (let j = 0; j < args.width; j++, n += 4) {
 85 			const pixels = extractPixels(t,i,j,Math.trunc(dimKer/2),args.height,args.width);
 86 			const R = newValue(ker,pixels,dimKer).R;
 87 			const G = newValue(ker,pixels,dimKer).G;
 88 			const B = newValue(ker,pixels,dimKer).B;
 89 			const A = newValue(ker,pixels,dimKer).A;
 90 			newImage[n] = R;
 91 			newImage[n+1] = G,
 92 			newImage[n+2] = B,
 93 			newImage[n+3] = A;
 94 		}
 95 	}
 96 	return newImage;
 97 }
