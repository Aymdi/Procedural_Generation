1 const { createImageData } = require('canvas');
  2 
  3 
  4 /*
  5  * A single color image
  6  * args entries : width, height, color
  7  */
  8 function mono(args) {
  9     if (typeof args.color === 'undefined') { args.color = [255,0,0,255]; }
 10 
 11     let image = createImageData(args.width, args.height);
 12     let n = 0;
 13     for (let x = 0; x < args.height; x++) {
 14 	for (let y = 0; y < args.width; y++, n+= 4) {
 15             for (let i = 0; i < 4; i++) {
 16                 image.data[n+i] = args.color[i];
 17             }
 18         }
 19     }
 20     return image;
 21 }
 22 
 23 
 24 exports.mono = mono;
